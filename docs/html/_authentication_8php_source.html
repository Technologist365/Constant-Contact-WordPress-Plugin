717  // Utility function for turning the Authorization: header into
718  // parameters, has to do some unescaping
719  // Can filter out any non-oauth parameters if needed (default behaviour)
720  public static function split_header($header, $only_allow_oauth_parameters = true) {
721  $pattern = '/(([-_a-z]*)=("([^"]*)"|([^,]*)),?)/';
722  $offset = 0;
723  $params = array();
724  while (preg_match($pattern, $header, $matches, PREG_OFFSET_CAPTURE, $offset) > 0) {
725  $match = $matches[0];
726  $header_name = $matches[2][0];

782  // This function takes a input like a=b&a=c&d=e and returns the parsed
783  // parameters like this
784  // array('a' => array('b','c'), 'd' => 'e')
785  public static function parse_parameters( $input ) {
786  if (!isset($input) || !$input) return array();
787 
788  $pairs = explode('&', $input);
789 
790  $parsed_parameters = array();
791  foreach ($pairs as $pair) {
792  $split = explode('=', $pair, 2);
793  $parameter = OAuthUtil::urldecode_rfc3986($split[0]);
794  $value = isset($split[1]) ? OAuthUtil::urldecode_rfc3986($split[1]) : '';
795 
796  if (isset($parsed_parameters[$parameter])) {
797  // We have already recieved parameter(s) with this name, so add to the list
798  // of parameters with this name
799 
800  if (is_scalar($parsed_parameters[$parameter])) {
801  // This is the first duplicate, so transform scalar (string) into an array
802  // so we can add the duplicates
803  $parsed_parameters[$parameter] = array($parsed_parameters[$parameter]);
804  }
805 
806  $parsed_parameters[$parameter][] = $value;
807  } else {
808  $parsed_parameters[$parameter] = $value;
809  }
810  }
811  return $parsed_parameters;

814  public static function build_http_query($params) {
815  if (!$params) return '';
816 
817  // Urlencode both keys and values
818  $keys = OAuthUtil::urlencode_rfc3986(array_keys($params));
819  $values = OAuthUtil::urlencode_rfc3986(array_values($params));
820  $params = array_combine($keys, $values);
821 
822  // Parameters are sorted by name, using lexicographical byte value ordering.
823  // Ref: Spec: 9.1.1 (1)
824  uksort($params, 'strcmp');
825 
826  $pairs = array();
827  foreach ($params as $parameter => $value) {
828  if (is_array($value)) {
829  // If two or more parameters share the same name, they are sorted by their value
830  // Ref: Spec: 9.1.1 (1)
831  natsort($value);
832  foreach ($value as $duplicate_value) {
833  $pairs[] = $parameter . '=' . $duplicate_value;
834  }
835  } else {
836  $pairs[] = $parameter . '=' . $value;
837  }
838  }
839  // For each parameter, the name is separated from the corresponding value by an '=' character (ASCII code 61)
840  // Each name-value pair is separated by an '&' character (ASCII code 38)
841  return implode('&', $pairs);
842  }
