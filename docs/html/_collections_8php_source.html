506  /**
507  * Upload an image to a folder
508  * @throws CTCTException - If file cannot me found or md5Hash cannot be created
509  * @param $imageLocation- Location of an image on disk
510  * @param $folderUrl - url of a folder to upload an image to
511  * @return Image
512  */
513  public function uploadImage($imageLocation, $folderUrl){
514  $imageName = substr(strrchr($imageLocation, '\\'), 1);
515  if($imageName == false) { $imageName = $imageLocation; }
516  $imageFormat = substr(strrchr($imageName, '.'), 1);
517  $boundary = "---------------------".substr(md5(rand(0,32000)), 0, 10);
518  try{
519  $file = file_get_contents($imageLocation);
520  $md5Hash = md5_file($imageLocation);
521  if(empty($file)){ throw new CTCTException('Unable to get file contents from: '.$imageLocation);}
522  if(empty($md5Hash)){ throw new CTCTException('Unable to create md5Hash from: '.$imageLocation);}
523  } catch (CTCTException $e){
524  $e->generateError('Error uploading image:');
525  }
526  switch($imageFormat){
527  case 'JPG':
528  $imageType = 'image/jpg';
529  break;
530  case 'GIF':
531  $imageType = 'image/gif';
532  break;
533  case 'PNG':
534  $imageType = 'image/png';
535  break;
536  default:
537  $imageType='image/jpeg';
538  }
539 
540  $xml = "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>
541  <atom:entry xmlns:atom=\"http://www.w3.org/2005/Atom\">
542  <atom:content>
543  <Image>
544  <FileName>".$imageName."</FileName>
545  <MD5Hash>".$md5Hash."</MD5Hash>
546  <Description />
547  </Image>
548  </atom:content>
549  </atom:entry>";
550 
551  $data = '--'.$boundary."\n";
552  $data .= "Content-Disposition: form-data; name=\"part1\"\n";
553  $data .= "Content-Type: application/atom+xml\n";
554  $data .= "Accept: application/atom+xml\n\n".$xml."\n";
555  $data .= "--".$boundary."\n";
556 
557  $data .= "Content-Disposition: form-data; name=\"part2\"; filename=\"".$imageName."\"\n";
558  $data .= "Content-Type: ".$imageType."g\n";
559  $data .= "Content-Transfer-Encoding: binary\n\n";
560  $data .= $file."\n";
561  $data .= "--".$boundary."--\n";
562 
563  $response = $this->CTCTRequest->makeRequest($folderUrl.'/images', 'POST', $data, "multipart/form-data; boundary=".$boundary);
564  $parsedResponse = simplexml_load_string($response['xml'], null, null, "http://www.w3.org/2005/Atom");
565  return new Image(Image::createStruct($parsedResponse));
566  }
