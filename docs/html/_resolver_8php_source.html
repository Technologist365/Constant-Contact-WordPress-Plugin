313  /* Net_DNS_Resolver::read_config {{{ */
314  /**
315  * Reads and parses a resolver configuration file
316  *
317  * @param string $file The name of the file to open and parse
318  */
319  function read_config($file)
320  {
321  if (is_readable($file)) {
322  if (! ($f = fopen($file, 'r'))) {
323  $this->error = "can't open $file";
324  }
325  }
326 
327  if (!is_resource($f)) {
328  $this->error = "can't open $file";
329  } else {
330  while (! feof($f)) {
331  $line = chop(fgets($f, 10240));
332  $line = ereg_replace('(.*)[;#].*', '\\1', $line);
333  if (ereg("^[ \t]*$", $line, $regs)) {
334  continue;
335  }
336  ereg("^[ \t]*([^ \t]+)[ \t]+([^ \t]+)", $line, $regs);
337  $option = $regs[1];
338  $value = $regs[2];
339 
340  switch ($option) {
341  case 'domain':
342  $this->domain = $regs[2];
343  break;
344  case 'search':
345  $this->searchlist[count($this->searchlist)] = $regs[2];
346  break;
347  case 'nameserver':
348  foreach (split(' ', $regs[2]) as $ns) {
349  $this->nameservers[count($this->nameservers)] = $ns;
350  }
351  break;
352  }
353  }
354  fclose($f);
355  }
356  }

433  /* Net_DNS_Resolver::nameservers() {{{ */
434  /**
435  * Gets or sets the nameservers to be queried.
436  *
437  * Returns the current nameservers if an array of new nameservers is not
438  * given as the argument OR sets the nameservers to the given nameservers.
439  *
440  * Nameservers not specified by ip address must be able to be resolved by
441  * the default settings of a new Net_DNS_Resolver.
442  *
443  * @access public
444  */
445  function nameservers($nsa = array())
446  {
447  $defres = new Net_DNS_Resolver();
448 
449  if (is_array($nsa)) {
450  $a = array();
451  foreach ($nsa as $ns) {
452  if (preg_match('/^(\d+(:?\.\d+){0,3})$/', $ns)) {
453  $a[] = ($ns == 0) ? '0.0.0.0' : $ns;
454  } else {
455  $names = array();
456  if (!preg_match('/\./', $ns)) {
457  if (!empty($defres->searchlist)) {
458  foreach ($defres->searchlist as $suffix) {
459  $names[] = $ns .'.' . $suffix;
460  }
461  } elseif (!empty($defres->domain)) {
462  $names[] = $ns .'.'. $defres->domain;
463  }
464  } else {
465  $names[] = $ns;
466  }
467  $packet = $defres->search($ns);
468  if (is_object($packet)) {
469  $addresses = $this->cname_addr($names, $packet);
470  foreach ($addresses as $b) {
471  $a[] = $b;
472  }
473  $a = array_unique($a);
474  }
475  }
476  }
477  if (count($a)) {
478  $this->nameservers = $a;
479  }
480  }
481  return $this->nameservers;
482  }

506  /* Net_DNS_Resolver::search() {{{ */
507  /**
508  * Searches nameservers for an answer
509  *
510  * Goes through the search list and attempts to resolve name based on
511  * the information in the search list.
512  *
513  * @param string $name The name (LHS) of a resource record to query.
514  * @param string $type The type of record to query.
515  * @param string $class The class of record to query.
516  * @return mixed an object of type Net_DNS_Packet on success,
517  * or false on failure.
518  * @see Net_DNS::typesbyname(), Net_DNS::classesbyname()
519  * @access public
520  */
521  function search($name, $type = 'A', $class = 'IN')
522  {
523  /*
524  * If the name looks like an IP address then do an appropriate
525  * PTR query.
526  */
527  if (preg_match('/^(\d+)\.(\d+)\.(\d+)\.(\d+)$/', $name, $regs)) {
528  $name = $regs[4].'.'.$regs[3].'.'.$regs[2].'.'.$regs[1].'.in-addr.arpa.';
529  $type = 'PTR';
530  }
531 
532  /*
533  * If the name contains at least one dot then try it as is first.
534  */
535  if (strstr($name, '.')) {
536  if ($this->debug) {
537  echo ";; search($name, $type, $class)\n";
538  }
539  $ans = $this->query($name, $type, $class);
540  if (is_object($ans) && ($ans->header->ancount > 0)) {
541  return $ans;
542  }
543  }
544 
545  /*
546  * If the name does not end in a dot then apply the search list.
547  */
548  $domain = '';
549  if ((! preg_match('/\.$/', $name)) && $this->dnsrch) {
550  foreach ($this->searchlist as $domain) {
551  $newname = "$name.$domain";
552  if ($this->debug) {
553  echo ";; search($newname, $type, $class)\n";
554  }
555  $ans = $this->query($newname, $type, $class);
556  if (is_object($ans) && ($ans->header->ancount > 0)) {
557  return $ans;
558  }
559  }
560  }
561 
562  /*
563  * Finally, if the name has no dots then try it as is.
564  */
565  if (strpos($name, '.') === false) {
566  if ($this->debug) {
567  echo ";; search($name, $type, $class)\n";
568  }
569  $ans = $this->query($name.'.', $type, $class);
570  if (is_object($ans) && ($ans->header->ancount > 0)) {
571  return $ans;
572  }
573  }
574 
575  /*
576  * No answer was found.
577  */
578  return false;
579  }

582  /* Net_DNS_Resolver::rawQuery() {{{ */
583  /**
584  * Queries nameservers for an answer
585  *
586  * Queries the nameservers listed in the resolver configuration for an
587  * answer to a question packet.
588  *
589  * @param string $name The name (LHS) of a resource record to query.
590  * @param string $type The type of record to query.
591  * @param string $class The class of record to query.
592  * @return mixed an object of type Net_DNS_Packet, regardless of whether the packet
593  * has an answer or not
594  * @see Net_DNS::typesbyname(), Net_DNS::classesbyname()
595  * @access public
596  */
597  function rawQuery($name, $type = 'A', $class = 'IN')
598  {
599  /*
600  * If the name does not contain any dots then append the default domain.
601  */
602  if ((strchr($name, '.') < 0) && $this->defnames) {
603  $name .= '.' . $this->domain;
604  }
605 
606  /*
607  * If the name looks like an IP address then do an appropriate
608  * PTR query.
609  */
610  if (preg_match('/^(\d+)\.(\d+)\.(\d+)\.(\d+)$/', $name, $regs)) {
611  $name = $regs[4].'.'.$regs[3].'.'.$regs[2].'.'.$regs[1].'.in-addr.arpa.';
612  $type = 'PTR';
613  }
614 
615  if ($this->debug) {
616  echo ";; query($name, $type, $class)\n";
617  }
618  $packet = new Net_DNS_Packet($this->debug);
619  $packet->buildQuestion($name, $type, $class);
620  $packet->header->rd = $this->recurse;
621  $ans = $this->send($packet);
622  return $ans;
623  }

651  /* Net_DNS_Resolver::send($packetORname, $qtype = '', $qclass = '') {{{ */
652  /**
653  * Sends a packet to a nameserver
654  *
655  * Determines the appropriate communication method (UDP or TCP) and
656  * sends a DNS packet to a nameserver. Use of the this function
657  * directly is discouraged. $packetORname should always be a properly
658  * formatted binary DNS packet. However, it is possible to send a
659  * query here and bypass Net_DNS_Resolver::query()
660  *
661  * @param string $packetORname A binary DNS packet stream or a
662  * hostname to query
663  * @param string $qtype This should not be used
664  * @param string $qclass This should not be used
665  * @return object Net_DNS_Packet An answer packet object
666  */
667  function send($packetORname, $qtype = '', $qclass = '')
668  {
669  $packet = $this->make_query_packet($packetORname, $qtype, $qclass);
670  $packet_data = $packet->data();
671 
672  if ($this->usevc != 0 || strlen($packet_data > 512)) {
673  $ans = $this->send_tcp($packet, $packet_data);
674  } else {
675  $ans = $this->send_udp($packet, $packet_data);
676 
677  if ($ans && $ans->header->tc && $this->igntc != 0) {
678  if ($this->debug) {
679  echo ";;\n;; packet truncated: retrying using TCP\n";
680  }
681  $ans = $this->send_tcp($packet, $packet_data);
682  }
683  }
684  return $ans;
685  }

719  /**
720  * Sends a packet via TCP to the list of name servers.
721  *
722  * @param string $packet A packet object to send to the NS list
723  * @param string $packet_data The data in the packet as returned by
724  * the Net_DNS_Packet::data() method
725  * @return object Net_DNS_Packet Returns an answer packet object
726  * @see Net_DNS_Resolver::send_udp(), Net_DNS_Resolver::send()
727  */
728  function send_tcp($packet, $packet_data)
729  {
730  if (! count($this->nameservers)) {
731  $this->errorstring = 'no nameservers';
732  if ($this->debug) {
733  echo ";; ERROR: send_tcp: no nameservers\n";
734  }
735  return null;
736  }
737  $timeout = $this->tcp_timeout;
738 
739  foreach ($this->nameservers as $ns) {
740  $dstport = $this->port;
741  if ($this->debug) {
742  echo ";; send_tcp($ns:$dstport)\n";
743  }
744  $sock_key = "$ns:$dstport";
745  if (isset($this->sockets[$sock_key]) && is_resource($this->sockets[$sock_key])) {
746  $sock = &$this->sockets[$sock_key];
747  } else {
748  if (! ($sock = @fsockopen($ns, $dstport, $errno,
749  $errstr, $timeout))) {
750  $this->errorstring = 'connection failed';
751  if ($this->debug) {
752  echo ";; ERROR: send_tcp: connection failed: $errstr\n";
753  }
754  continue;
755  }
756  $this->sockets[$sock_key] = $sock;
757  unset($sock);
758  $sock = &$this->sockets[$sock_key];
759  }
760  $lenmsg = pack('n', strlen($packet_data));
761  if ($this->debug) {
762  echo ';; sending ' . strlen($packet_data) . " bytes\n";
763  }
764 
765  if (($sent = fwrite($sock, $lenmsg)) == -1) {
766  $this->errorstring = 'length send failed';
767  if ($this->debug) {
768  echo ";; ERROR: send_tcp: length send failed\n";
769  }
770  continue;
771  }
772 
773  if (($sent = fwrite($sock, $packet_data)) == -1) {
774  $this->errorstring = 'packet send failed';
775  if ($this->debug) {
776  echo ";; ERROR: send_tcp: packet data send failed\n";
777  }
778  }
779 
780  socket_set_timeout($sock, $timeout);
781  $buf = fread($sock, 2);
782  $e = socket_get_status($sock);
783  /* If $buf is empty, we want to supress errors
784  long enough to reach the continue; down the line */
785  $len = @unpack('nint', $buf);
786  $len = @$len['int'];
787  if (!$len) {
788  continue;
789  }
790  $buf = fread($sock, $len);
791  $actual = strlen($buf);
792  $this->answerfrom = $ns;
793  $this->answersize = $len;
794  if ($this->debug) {
795  echo ";; received $actual bytes\n";
796  }
797  if ($actual != $len) {
798  $this->errorstring = "expected $len bytes, received $buf";
799  if ($this->debug) {
800  echo ';; send_tcp: ' . $this->errorstring;
801  }
802  continue;
803  }
804 
805  $ans = new Net_DNS_Packet($this->debug);
806  if (is_null($ans->parse($buf))) {
807  continue;
808  }
809  $this->errorstring = $ans->header->rcode;
810  $ans->answerfrom = $this->answerfrom;
811  $ans->answersize = $this->answersize;
812  return $ans;
813  }

831  function send_udp_no_sock_lib($packet, $packet_data)

836  /*
837  * PHP doesn't have excellent socket support as of this writing.

908  continue;
909  }
910  $this->errorstring = $ans->header->rcode;
911  $ans->answerfrom = $this->answerfrom;
912  $ans->answersize = $this->answersize;
913  return $ans;
914  }
915  }
916  // Sleep another 1/100th of a second... this sucks...
917  usleep(1000);
918  }
919 
920  }
921 
922  $this->errorstring = 'query timed out';
923  return null;
924  }

929  /**
930  * Sends a packet via UDP to the list of name servers.
931  *
932  * This function sends a packet to a nameserver. It is called by
933  * send_udp if the sockets PHP extension is compiled into PHP.
934  *
935  * @param string $packet A packet object to send to the NS list
936  * @param string $packet_data The data in the packet as returned by
937  * the Net_DNS_Packet::data() method
938  * @return object Net_DNS_Packet Returns an answer packet object
939  * @see Net_DNS_Resolver::send_tcp(), Net_DNS_Resolver::send(),
940  * Net_DNS_Resolver::send_udp(), Net_DNS_Resolver::send_udp_no_sock_lib()
941  */
942  function send_udp_with_sock_lib($packet, $packet_data)
943  {
944  $retrans = $this->retrans;
945  $timeout = $retrans;
946 
947  //$w = error_reporting(0);
948  $ctr = 0;
949  // Create a socket handle for each nameserver
950  foreach ($this->nameservers as $nameserver) {
951  if ((($sock[$ctr++] = socket_create(AF_INET, SOCK_DGRAM, SOL_UDP))) &&
952  socket_connect($sock[$ctr-1], $nameserver, $this->port))
953  {
954  $peerhost[$ctr-1] = $nameserver;
955  $peerport[$ctr-1] = $this->port;
956  socket_set_nonblock($sock[$ctr-1]);
957  } else {
958  $ctr--;
959  }
960  }
961  //error_reporting($w);
962 
963  if ($ctr == 0) {
964  $this->errorstring = 'no nameservers';
965  return null;
966  }
967  // Try each nameserver up to $this->retry times
968  for ($i = 0; $i < $this->retry; $i++) {
969  if ($i != 0) {
970  // Set the timeout for each retry based on the number of
971  // nameservers there is a connected socket for.
972  $retrans *= 2;
973  $timeout = (int) ($retrans / $ctr);
974  }
975  // Make sure the timeout is at least 1 second
976  if ($timeout < 1) {
977  $timeout = 1;
978  }
979 
980  // Try each nameserver
981  foreach ($sock as $k => $s) {
982  if ($this->debug) {
983  echo "\n;; send_udp(" . $peerhost[$k] . ':' . $peerport[$k] . '): sending ' . strlen($packet_data) . " bytes\n";
984  }
985 
986  if (! socket_write($s, $packet_data)) {
987  if ($this->debug) {
988  echo ";; send error\n";
989  }
990  }
991 
992  $set = array($s);
993  if ($this->debug) {
994  echo ";; timeout set to $timeout seconds\n";
995  }
996  $changed = socket_select($set, $w = null, $e = null, $timeout);
997  if ($changed) {
998  // Test to see if the connection was refused. Linux servers will send
999  // an ICMP message which will cause the client's next system call to
1000  // return ECONNREFUSED if the server is not listening on the ip:port queried
1001  if (socket_get_option($s, SOL_SOCKET, SO_ERROR) == SOCKET_ECONNREFUSED) {
1002  // Unix socket connection was refused
1003  if ($this->debug) {
1004  echo ';; connection to ' . $peerhost[$k] . ':' . $peerport[$k] . " was refused\n";
1005  }
1006  // Try the next server.
1007  continue;
1008  }
1009 
1010  // Read the response
1011  $buf = @socket_read($s, 512);
1012  if ($buf === false) {
1013  // No data could be read from socket
1014  if ($this->debug) {
1015  echo ';; no data could be read from ' . $peerhost[$k] . ':' . $peerport[$k] . "\n";
1016  echo ';; socket_error: ' . socket_strerror(socket_last_error()) . "\n";
1017  }
1018  // Reset the non-specific socket error status
1019  socket_clear_error();
1020  // Try the next server.
1021  continue;
1022  }
1023 
1024  $this->answerfrom = $peerhost[$k];
1025  $this->answersize = strlen($buf);
1026  if ($this->debug) {
1027  echo ';; answer from ' . $peerhost[$k] . ':' .
1028  $peerport[$k] . ': ' . strlen($buf) . " bytes\n";
1029  }
1030  $ans = new Net_DNS_Packet($this->debug);
1031  if ($ans->parse($buf)) {
1032  if ($ans->header->qr != '1') {
1033  // Ignore packet if it is not a response
1034  continue;
1035  } elseif ($ans->header->id != $packet->header->id) {
1036  // Ignore packet if the response id does not match the query id
1037  continue;
1038  } else {
1039  // Return the DNS response packet
1040  $this->errorstring = $ans->header->rcode;
1041  $ans->answerfrom = $this->answerfrom;
1042  $ans->answersize = $this->answersize;
1043  return $ans;
1044  }
1045  }
1046  } elseif ($this->debug) {
1047  echo ";; query to ". $peerhost[$k] . ':' . $peerport[$k] . " timed out\n";
1048  }
1049  }
1050  }
1051  $this->errorstring = 'query timed out';
1052  return null;
1053  }
