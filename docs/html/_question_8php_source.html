39  /* class constructor Net_DNS_Question($qname, $qtype, $qclass) {{{ */
40  function Net_DNS_Question($qname, $qtype, $qclass)
41  {
42  $qtype = !is_null($qtype) ? strtoupper($qtype) : 'ANY';
43  $qclass = !is_null($qclass) ? strtoupper($qclass) : 'ANY';
44 
45  // Check if the caller has the type and class reversed.
46  // We are not that kind for unknown types.... :-)
47  if ( ( is_null(Net_DNS::typesbyname($qtype)) ||
48  is_null(Net_DNS::classesbyname($qtype)) )
49  && !is_null(Net_DNS::classesbyname($qclass))
50  && !is_null(Net_DNS::typesbyname($qclass)))
51  {
52  list($qtype, $qclass) = array($qclass, $qtype);
53  }
54  $qname = preg_replace(array('/^\.+/', '/\.+$/'), '', $qname);
55  $this->qname = $qname;
56  $this->qtype = $qtype;
57  $this->qclass = $qclass;
58  }
59  /* }}} */
60  /* Net_DNS_Question::display() {{{*/
61  function display()
62  {
63  echo $this->string() . "\n";
64  }
65 
66  /*}}}*/
67  /* Net_DNS_Question::string() {{{*/
68  function string()
69  {
70  return $this->qname . ".\t" . $this->qclass . "\t" . $this->qtype;
71  }

75  function data($packet, $offset)
76  {
